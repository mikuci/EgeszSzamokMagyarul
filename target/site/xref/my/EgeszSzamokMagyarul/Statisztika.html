313     /**
314      * "Statisztika betöltése" gomb eseménye, kattintás esetén. Menti a beállításokat és
315      * leállítja a programot.
316      *
317      * @param evt esemény
318      */
319     private void jLoadStaticsButtonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jLoadStaticsButtonActionPerformed
320         final String default_ = "nincs";
321         final Statisztika This = this;
322         SwingWorker<Void, Void> worker = new SwingWorker<Void, Void>() {
323             @Override
324             protected Void doInBackground() throws Exception {
325                 if (ProgramManager.loadStaticsInXml()) {
326                     if (ProgramManager.isNumber(ProgramManager.getT(0, 0))) {
327                         if (Integer.valueOf(ProgramManager.getT(0, 0)) > 0) {
328                             jLabel_A1.setText(ProgramManager.getT(0, 0) + " db.");
329                             jLabel_A2.setText(ProgramManager.getT(0, 1) + " db.");
330                             jLabel_A3.setText(ProgramManager.getT(0, 2) + " ms.");
331                         } else {
332                             jLabel_A1.setText(default_);
333                             jLabel_A2.setText(default_);
334                             jLabel_A3.setText(default_);
335                         }
336                     }
337 
338                     if (ProgramManager.isNumber(ProgramManager.getT(1, 0))) {
339                         if (Integer.valueOf(ProgramManager.getT(1, 0)) > 0) {
340                             jLabel_B1.setText(ProgramManager.getT(1, 0) + " db.");
341                             jLabel_B2.setText(ProgramManager.getT(1, 1) + " db.");
342                             jLabel_B3.setText(ProgramManager.getT(1, 2) + " ms.");
343                         } else {
344                             jLabel_B1.setText(default_);
345                             jLabel_B2.setText(default_);
346                             jLabel_B3.setText(default_);
347                         }
348                     }
349                 } else {
350                     Fomenu.fomenu.setVisible(true);
351                     This.dispose();
352                 }
353                 return null;
354             }
355 
356         };
357 
358         worker.execute();
359     }//GEN-LAST:event_jLoadStaticsButtonActionPerformed
