222     /**
223 	 * Egy adott helyi érteken lévő, egész szám, magyar nyelvű nevét adja vissza.
224 	 * 
225      * @param HelyiÉrték szám HelyiÉrtéke
226      * @param Érték egész típus
227      * @param EgyedűlÁllóNév igaz lehet például a kettes szám esetén
228      * <ul>
229 	 *     <li>{@code "kettő"}, ha pédául a helyi értéke 0,</li>
230 	 *     <li>{@code "kétezer"}, ha azonban értéke 3.</li>
231 	 * </ul>
232      * @param t SzamNev típusú tömb, melyben megtalálhatóak a számok magyar nyelvű leírása
233      * @return
234      * <ul>
235      *     <li>{@code az egész szám neve az adott HelyiÉrtéken}, ha megtalálja az aktuális számot,</li>
236      *     <li>{@code "ismeretlen szám"}, egyébként.</li>
237      * </ul>
238      */
239     private String kereso(int HelyiÉrték, double Érték, boolean EgyedűlÁllóNév, SzamNev[] t){
240         try{
241             for (SzamNev t1 : t) {
242                 if (t1.getÉrték() == Érték && t1.getHelyiÉrték() == HelyiÉrték) {
243                     if (t1.isMásAzEgyedülÁllóNév() && EgyedűlÁllóNév) {
244                         return t1.getEgyedülÁllóNév();
245                     }
246                     return t1.getNév();
247                 } else if ((t1.getÉrtékTízHatványban() == Érték && t1.isÉrtékTízHatványbanVan()) && t1.getHelyiÉrték() == HelyiÉrték) {
248                     if (t1.isMásAzEgyedülÁllóNév() && EgyedűlÁllóNév) {
249                         return t1.getEgyedülÁllóNév();
250                     }
251                     return t1.getNév();
252                 }
253             }
254         }catch(Exception e){
255             System.out.println(e.getMessage());
256         }
257         return "imeretlen_szám";
258     }

260     /**
261      * Azonos helyi értéken lévő egész számok magyar nyelvű nevét adja vissza.
262      * 
263      * @param HelyiÉrték szám HelyiÉrtéke
264      * @param Érték egész szám
265      * @param UtolsóASajátHelyiÉrtékén logikai igazság érték, amely 
266      * <ul>
267 	 *     <li>{@code true}, ha ezen a HelyiÉrtéken, ez az utolsó nem nulla szám,</li>
268 	 *     <li>{@code false}, egyébként.</li>
269 	 * </ul>
270      * @param UtolsóSzám logikai igazság érték, amely 
271      * <ul>
272 	 *     <li>{@code true}, ha ez az utolsó nem nulla számjegy,</li>
273 	 *     <li>{@code false}, egyébként.</li>
274 	 * </ul>
275      * @param t SzamNev típusú tömb, melyben megtalálhatóak a számok magyar nyelvű leírása
276      * @param t2 SzamNev típusú tömb, melyben megtalálhatóak a számok magyar nyelvű leírása, ami -dik toldalékkal van kiegészítve
277      * @param dik  logikai igazság érték, amely 
278      * <ul>
279      *     <li>{@code true}, ha kell -dik toldalék a szám végére,</li>
280      *     <li>{@code false}, egyébként.</li>
281      * </ul>
282      * @return azonos helyi értéken lévő egész számok magyar nyelvű neve
283      * @see #Kereső(int, double, boolean, my.EgészSzámokMagyarul.SzámNév[]) 
284      */
285     private String neve(int HelyiÉrték, int Érték, boolean UtolsóASajátHelyiÉrtékén, boolean UtolsóSzám, SzamNev[] t, SzamNev[] t2, boolean dik){
286         StringBuilder sb = new StringBuilder();
287         double VÉrték;
288         
289         try{
290             if(HelyiÉrték > 3)
291                 VÉrték = HelyiÉrték;
292             else
293                 VÉrték = tizHatvany(HelyiÉrték);
294 
295             if(Érték == 0 && !UtolsóSzám)
296                 return "";
297 
298             if(HelyiÉrték < 2)
299             {
300                 if(UtolsóSzám || Érték != 0)
301                 {
302                     if(dik && UtolsóSzám)
303                         sb.append(kereso(HelyiÉrték, VÉrték * Érték, UtolsóSzám, t2));
304                     else
305                         sb.append(kereso(HelyiÉrték, VÉrték * Érték, UtolsóSzám, t));
306                 }
307             }
308             else if(HelyiÉrték >= 2)
309             {
310                 if(HelyiÉrték % 3 == 0 || HelyiÉrték == 2){
311                     if(Érték > 0)
312                         sb.append(kereso(0, Érték, false, t));
313                     if(dik && UtolsóSzám)
314                         sb.append(kereso(HelyiÉrték, VÉrték, UtolsóSzám, t2));
315                     else sb.append(kereso(HelyiÉrték, VÉrték, UtolsóSzám, t));
316 
317                     if(HelyiÉrték > 3 && Érték > 0 && HelyiÉrték % 3 == 0 && !UtolsóSzám)
318                         sb.append("-");
319                     else if(HelyiÉrték == 3 && Érték > 1 && !UtolsóSzám)
320                         sb.append("-");
321                 }
322                 else
323                 {
324                     int marad = HelyiÉrték % 3, mehet = HelyiÉrték - marad;
325                     sb.append(neve(marad, Érték, UtolsóASajátHelyiÉrtékén, false, t, t2, false));
326                     if(UtolsóASajátHelyiÉrtékén){
327                         sb.append(neve(mehet, 1, UtolsóASajátHelyiÉrtékén, UtolsóSzám, t, t2, dik));
328                         sb.append("-");
329                     }
330                 }
331             }
332         }catch(Exception e){
333             System.out.println(e.getMessage());
334         }
335         return sb.toString();
336     }

351     private String egeszSzamToSztring(boolean dik) {
352 		StringBuilder szöveg = new StringBuilder();
353                 boolean Utolsó = false;
354 		
355                 int[] helyiértékek = helyiErtek.getHelyiÉrtékek();
356                 boolean[] UtolsóA = helyiErtek.getUtolsóASajátHelyiÉrtékén();
357                 
358 		for(int i = helyiErtek.getlength(); i >= helyiErtek.getUtolsóSzámHelyiÉrtéke(); i--)
359                 {
360                     if(i == helyiErtek.getUtolsóSzámHelyiÉrtéke())
361                         Utolsó = true;
362                     
363                     String ss = neve(i, helyiértékek[i], UtolsóA[i], Utolsó, szamnevek, szamnevek_dik, dik);
364                     szöveg.append(ss);
365                 }
366 		
367 		return szöveg.toString();
368 	}
