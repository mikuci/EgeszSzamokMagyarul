119     /**
120      * Beolvassa a kiválasztott xml állományt és visszadja a keresett elemet.
121      *
122      * @param ElementsByTagName Keresett elem neve.
123      * @return az adott elem
124      * @throws ParserConfigurationException {@inheritDoc}
125      * @throws SAXException {@inheritDoc}
126      * @throws IOException {@inheritDoc}
127      * @see #xmlFile
128      */
129     @SuppressWarnings("empty-statement")
130     private Element readXmlFile(String ElementsByTagName) throws ParserConfigurationException, SAXException, IOException {
131         String fileName = "";
132 
133         if (xmlFile == XMLFile.CONFIGURATION) {
134             fileName = configurationXMLPath;
135         } else if (xmlFile == XMLFile.STATICS) {
136             fileName = staticsXMLPath;
137         }
138 
139         File fXmlFile = new File(fileName);
140         DocumentBuilderFactory dbFactory = DocumentBuilderFactory.newInstance();
141         DocumentBuilder dBuilder = dbFactory.newDocumentBuilder();
142         Document doc = dBuilder.parse(fXmlFile);
143         doc.getDocumentElement().normalize();
144         NodeList nList = doc.getElementsByTagName(ElementsByTagName);
145 
146         for (int temp = 0; temp < nList.getLength(); temp++) {
147             Node nNode = nList.item(temp);
148             if (nNode.getNodeType() == Node.ELEMENT_NODE) {
149                 Element eElement = (Element) nNode;
150                 if (eElement.getAttribute("userName").equals(ProgramManager.getUserName())) {
151                     return eElement;
152                 }
153             }
154         }
155 
156         return null;
157     }

159     /**
160      * Egy kiválasztott elemből visszadja keresett leszármazott elemet.
161      *
162      * @param element egy xml elem
163      * @return keresett leszármazott elem
164      * @see #Attribute
165      * @see #Value
166      */
167     @SuppressWarnings("empty-statement")
168     private Element readXmlElement(Element element) {
169         NodeList nList = (NodeList) element;
170         for (int temp = 0; temp < nList.getLength(); temp++) {
171             Node nNode = nList.item(temp);
172             if (nNode.getNodeType() == Node.ELEMENT_NODE) {
173                 Element eElement = (Element) nNode;
174                 if (eElement.getAttribute(Attribute).equals(Value)) {
175                     return eElement;
176                 }
177             }
178         }
179 
180         return null;
181     }

183     /**
184      * Betölti xml fájlból, egy éréket az elem neve alapján.
185      *
186      * @param thisElementName elem neve
187      * @return elem értéke
188      * @throws TransformerException {@inheritDoc}
189      * @throws TransformerConfigurationException {@inheritDoc}
190      * @throws ParserConfigurationException {@inheritDoc}
191      * @throws SAXException {@inheritDoc}
192      * @throws IOException {@inheritDoc}
193      */
194     public String loadXmlvalue(String thisElementName) throws TransformerException, TransformerConfigurationException, ParserConfigurationException, SAXException, IOException {
195         Element eElement = readXmlFile(parentElementName);
196         if (eElement == null) {
197             if (xmlFile == XMLFile.CONFIGURATION) {
198                 saveToXmlWithConfig("FőmenüBeállítások");
199             }
200             eElement = readXmlFile(parentElementName);
201         }
202 
203         return eElement.getElementsByTagName(thisElementName).item(0).getTextContent();
204     }

210      * @return a megadott elem értéke

212      * @throws org.xml.sax.SAXException {@inheritDoc}
213      * @throws java.io.IOException {@inheritDoc}

215     public String loadXmlvalueByAttribute(String thisElementName) throws ParserConfigurationException, SAXException, IOException {
216         Element eElement = readXmlFile(parentElementName);
217         eElement = readXmlElement(eElement);
218         return eElement.getElementsByTagName(thisElementName).item(0).getTextContent();
219     }

221     /**
222      * Kimenti egy xml fájlba az aktuális felhasználó beállításokat.
223      *
224      * @param ElementName az xml elem neve, ami tartalmazza a beállításokat
225      * @throws javax.xml.transform.TransformerConfigurationException
226      * {@inheritDoc}
227      * @throws org.xml.sax.SAXException {@inheritDoc}
228      * @throws javax.xml.parsers.ParserConfigurationException {@inheritDoc}
229      * @throws java.io.IOException {@inheritDoc}
230      */
231     @SuppressWarnings("empty-statement")
232     public void saveToXmlWithConfig(String ElementName) throws TransformerConfigurationException, TransformerException, ParserConfigurationException, SAXException, IOException {
233         String fileName = configurationXMLPath;
234 
235         Boolean ok = false;
236         File fXmlFile = new File(fileName);
237         Document doc = DocumentBuilderFactory.newInstance().newDocumentBuilder().parse(fXmlFile);
238         doc.setXmlStandalone(true);
239 
240         doc.getDocumentElement().normalize();
241         NodeList nList = doc.getElementsByTagName(ElementName);
242 
243         for (int temp = 0; temp < nList.getLength(); temp++) {
244             Node nNode = nList.item(temp);
245             if (nNode.getNodeType() == Node.ELEMENT_NODE) {
246                 Element eElement = (Element) nNode;
247                 if (eElement.getAttribute("userName").equals(ProgramManager.getUserName())) {
248                     eElement.getElementsByTagName("idő").item(0).setTextContent(Integer.toString(ProgramManager.getIdo()));
249                     eElement.getElementsByTagName("maxszám").item(0).setTextContent(Integer.toString(ProgramManager.getMaxszam()));
250                     eElement.getElementsByTagName("dik").item(0).setTextContent(Boolean.toString(ProgramManager.isDik()));
251 
252                     ok = true;
253                 }
254             }
255         }
256 
257         if (!ok) {
258             Element rootElement = doc.getDocumentElement();
259             Element staff = doc.createElement(ElementName);
260             staff.setAttribute("userName", ProgramManager.getUserName());
261             rootElement.appendChild(staff);
262 
263             Element elem;
264             elem = doc.createElement("idő");
265             elem.appendChild(doc.createTextNode(Integer.toString(ProgramManager.getIdo())));
266             staff.appendChild(elem);
267 
268             elem = doc.createElement("maxszám");
269             elem.appendChild(doc.createTextNode(Integer.toString(ProgramManager.getMaxszam())));
270             staff.appendChild(elem);
271 
272             elem = doc.createElement("dik");
273             elem.appendChild(doc.createTextNode(Boolean.toString(ProgramManager.isDik())));
274             staff.appendChild(elem);
275         }
276 
277         DOMSource source = new DOMSource(doc);
278         StreamResult result = new StreamResult(fXmlFile);
279         TransformerFactory.newInstance().newTransformer().transform(source, result);
280     }

401      * @throws TransformerException  {@inheritDoc}
402      */
403     private static void createDefaultXML(XMLFile xml) throws ParserConfigurationException, SAXException, IOException, TransformerConfigurationException, TransformerException {
404         File fXmlFile = null;
405 
406         if (xml == XMLFile.CONFIGURATION) {
407             fXmlFile = new File(configurationXMLPath);
408         } else if (xml == XMLFile.STATICS) {
409             fXmlFile = new File(staticsXMLPath);
410         }
411 
412         if (!fXmlFile.exists()) {
413             Document doc = DocumentBuilderFactory.newInstance().newDocumentBuilder().newDocument();
414             doc.createDocumentFragment();
415             if (xml == XMLFile.CONFIGURATION) {
416                 Element rootElement = doc.createElement("Beallítások");
417                 doc.appendChild(rootElement);
418                 Element staff = doc.createElement("FőmenüBeállítások");
419                 staff.setAttribute("userName", "default");
420                 rootElement.appendChild(staff);
421 
422                 Element elem;
423                 elem = doc.createElement("idő");
424                 elem.appendChild(doc.createTextNode(Integer.toString(ProgramManager.getIdo())));
425                 staff.appendChild(elem);
426 
427                 elem = doc.createElement("maxszám");
428                 elem.appendChild(doc.createTextNode(Integer.toString(ProgramManager.getMaxszam())));
429                 staff.appendChild(elem);
430 
431                 elem = doc.createElement("dik");
432                 elem.appendChild(doc.createTextNode(Boolean.toString(ProgramManager.isDik())));
433                 staff.appendChild(elem);
434             }
435             if (xml == XMLFile.STATICS) {
436                 Element rootElement = doc.createElement("Statisztikák");
437                 doc.appendChild(rootElement);
438                 Element staff2 = doc.createElement("Statisztika");
439                 staff2.setAttribute("userName", "default");
440                 rootElement.appendChild(staff2);
441 
442                 Element staff = doc.createElement("Gyakorlás");

444                 Attr attr = doc.createAttribute("id");
445                 attr.setValue(Integer.toString(1));
446                 staff.setAttributeNode(attr);
447 
448                 Element elem;
449                 elem = doc.createElement("feladatok");
450                 elem.appendChild(doc.createTextNode(Integer.toString(0)));
451                 staff.appendChild(elem);
452 
453                 elem = doc.createElement("helyes");
454                 elem.appendChild(doc.createTextNode(Integer.toString(0)));
455                 staff.appendChild(elem);
456 
457                 elem = doc.createElement("idő");
458                 elem.appendChild(doc.createTextNode(Integer.toString(ProgramManager.getIdo())));
459                 staff.appendChild(elem);
460             }
461             DOMSource source = new DOMSource(doc);
462             StreamResult result = new StreamResult(fXmlFile);
463             Transformer trans = TransformerFactory.newInstance().newTransformer();
464             //trans.setOutputProperty(OutputKeys.ENCODING, "UTF8");
465             trans.transform(source, result);
466         }
